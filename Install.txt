#####x11vnc server

sudo apt-get install x11vnc
x11vnc -storepasswd /home/pi/vncpasswd
                                  

/home/pi/.config/autostart/x11vnc.desktop:

[Desktop Entry]
Encoding=UTF-8
Type=Application
Name=X11VNC
Comment=
Exec=x11vnc -forever -usepw -display :0 -ultrafilexfer -rfbauth /home/pi/vncpasswd
StartupNotify=false
Terminal=false
Hidden=false
